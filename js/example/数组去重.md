## js实现数组去重
采用原型链方法和对象哈希实现

```javascript
var arr = [1, 1, 2, 2, 3, 3, "a", "a"]

Array.prototype.unique = function() {
    var obj = {},
        newArr = [],
        len = arr.length;
    for (var i = 0; i < len; i++) {
        if (!obj[arr[i]]) {
            obj[arr[i]] = "abc"
            newArr.push(arr[i]);
        }
    }
    return newArr;
}

console.log(arr.unique()) // [1, 2, 3, "a"]
```
